La funcin cuenta_digitos (nombre) que recibe el nombre del archivo de texto y regrese el nmero de dgitos contenidos en un archivo de texto.

Por ejemplo: Si en el archivo de texto tiene almacenado el texto:

"Luz" 1973 Zoila 2000 -Pedro- 2024

def cuenta_digitos(nombre):
    digitos = "0123456789"
    file = open(nombre, "r")
    contenido = file.read(1)
    cont = 0

    while contenido != "":
        if contenido in digitos:
            cont += 1
        contenido = file.read(1)
    file.close()
    return cont


def main():
    nombre_archivo = input("Ingresa el nombre del archivo: ")
    total_digitos = cuenta_digitos(nombre_archivo)
    print("El archivo %s contiene %s digitos." % (nombre_archivo, total_digitos))


main()

def cuenta_caracteres(nombre):
    file = open(nombre, "r")
    contenido = file.read(1)
    cont = 0

    while contenido != "":
        if contenido != " " and contenido != "\n":
            cont += 1
        contenido = file.read(1)
    file.close()
    return cont


def main():
    nombre_archivo = input("Ingresa el nombre del archivo: ")
    total_caracteres = cuenta_caracteres(nombre_archivo)
    print("El archivo %s contiene %s caracteres" % (nombre_archivo, total_caracteres))


main()
